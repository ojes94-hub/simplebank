## run postgres docker image 
    docker run --name postgres18 -p 5432:5432 -e POSTGRES_USER=root -e POSTGRES_PASSWORD=daniel@@1 -d postgres:18.1-alpine3.23

## Show docker containers
    docker ps

## Show docker images
    docker images

## Login and enter postgres enviroment
    docker exec -it postgres18 psql -U root

## Exit from postgres enviroment
    \q

## Get Logs of the docker postgres database
    docker logs postgres18
